import { Playground, Props } from 'docz'
import { Button, Icon, Boxer, buttonVariants, buttonSizes } from 'scaleway-ui'

# Button

## Variants and Sizes Table

Set variant and size using `variant` and `size` property.

<Playground>
  <Boxer my={1}>
    <div
      style={{
        display: 'flex',
        justifyContent: 'space-between',
        width: '100%',
      }}
    >
      {buttonSizes.map(size => (
        <div key={size} style={{ display: 'flex', flexDirection: 'column' }}>
          {buttonVariants.map(variant => (
            <div
              key={variant}
              style={{
                alignItems: 'center',
                display: 'flex',
                flexDirection: 'column',
                height: '130px',
                justifyContent: 'space-between',
                padding: '10px',
                width: '130px',
              }}
            >
              <div
                style={{
                  alignItems: 'center',
                  display: 'flex',
                  flex: 1,
                  fontSize: '10px',
                  fontWeight: 500,
                  letterSpacing: '2px',
                  textAlign: 'center',
                }}
              >
                {size} {variant}
              </div>
              <div
                style={{
                  alignItems: 'center',
                  display: 'flex',
                  flex: 1,
                }}
              >
                <Button size={size} variant={variant}>
                  BUTTON
                </Button>
              </div>
            </div>
          ))}
        </div>
      ))}
    </div>
  </Boxer>
</Playground>

## Variants

Set variant using `variant` property.

<Playground>
  <Boxer my={1}>
    {buttonVariants.map(variant => (
      <Button key={variant} variant={variant}>
        {variant}
      </Button>
    ))}
  </Boxer>
</Playground>

## Sizes

Set size using `size` property.

<Playground>
  <Boxer my={1}>
    {buttonSizes.map(size => (
      <Button key={size} size={size}>
        {size}
      </Button>
    ))}
  </Boxer>
</Playground>

## Disabled

Set disabled using `disabled` property.

<Playground>
  <Boxer my={1}>
    {buttonVariants.map(variant => (
      <Button key={variant} variant={variant} disabled>
        {variant} disabled
      </Button>
    ))}
  </Boxer>
</Playground>

## Progress

Add `ActivityIndicator` by setting `progress` property to `left` or `right`.

<Playground>
  <Boxer my={1}>
    {buttonSizes.map(size => (
      <Button key={`left-${size}`} size={size} progress="left">
        {size} left progress
      </Button>
    ))}
    {buttonSizes.map(size => (
      <Button key={`right-${size}`} size={size} progress="right">
        {size} right progress
      </Button>
    ))}
    {buttonSizes.map(size => (
      <Button key={size} size={size} progress />
    ))}
  </Boxer>
</Playground>

## Icons

Add icon using `icon`. You can specify the name of the icon or the icon itself.

<Playground>
  <Boxer my={1}>
    {buttonSizes.map(size => (
      <React.Fragment key={size}>
        <Button size={size} icon="lock">
          Icon + Text
        </Button>
        <Button size={size} icon="lock" />
      </React.Fragment>
    ))}
  </Boxer>
</Playground>

## Icons Positions

Add a position icon using `iconPosition`. You can specify whether you want it to the left or to the right of your text. It requires an `icon`.

<Playground>
  <Boxer my={1}>
    {['left', 'right'].map(position => (
      <React.Fragment key={position}>
        <Button iconPosition={position} icon="lock">
          {position}
        </Button>
      </React.Fragment>
    ))}
  </Boxer>
</Playground>

## Extend

Make button extensible using `extend` prop. It requires a `icon`.

<Playground>
  <Boxer my={1}>
    {buttonSizes.map(size => (
      <Button key={size} extend size={size} icon="plus">
        {size} extend
      </Button>
    ))}
  </Boxer>
</Playground>

## Link delegation

Using the `to` props, link behavior is delegated to [Link Component](docs-components-link).

<Playground>
  <Boxer my={1}>
    <Button to="https://scaleway.com" target="_blank">
      Scaleway
    </Button>
  </Boxer>
</Playground>
