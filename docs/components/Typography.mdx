import { Playground, Props } from 'docz'
import { Box, Boxer, Typography, typographyVariants } from 'scaleway-ui'

# Typography

## Variants

Set variant using `variant` property.

<Playground>
  <Boxer my={1}>
    {typographyVariants.map(variant => (
      <Typography key={variant} variant={variant}>
        {variant}
      </Typography>
    ))}
  </Boxer>
</Playground>

## Ellipsis

Force the text on a single line using `ellipsis` prop.

<Playground>
  <strong>Without ellipsis</strong>
  <Box width={500} mt={1} mb={2}>
    <Typography>
      This text is quite long. Lorem ipsum dolor sit amet, consectetur
      adipisicing elit, sed do eiusmod tempor incididunt ut labore et dolore
      magna aliqua.
    </Typography>
  </Box>
  <strong>With ellipsis (a tooltip is displayed on hover)</strong>
  <Box width={500} mt={1} mb={2}>
    <Typography ellipsis>
      This text is quite long. Lorem ipsum dolor sit amet, consectetur
      adipisicing elit, sed do eiusmod tempor incididunt ut labore et dolore
      magna aliqua.
    </Typography>
  </Box>
</Playground>
