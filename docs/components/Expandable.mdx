import { Playground } from 'docz'
import { Expandable, Touchable, Typography, Icon, Toggler } from 'scaleway-ui'

# Expandable

Expandable container.

## Basic

Expand using `opened`. Set `height` with the approximate height of the container to have a beautiful animation.

<Playground>
  <Toggler>
    {([toggled, toggle]) => (
      <>
        <Touchable onClick={() => toggle()}>
          <Typography variant="lead-block" my={0}>
            Add Some Advanced Options?
            <Icon name={toggled ? 'minus-box-outline' : 'plus-box-outline'} />
          </Typography>
        </Touchable>
        <Expandable height={24} opened={toggled}>
          Advanced options
        </Expandable>
      </>
    )}
  </Toggler>
</Playground>
