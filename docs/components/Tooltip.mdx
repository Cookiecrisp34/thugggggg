import { Box, Tooltip } from 'scaleway-ui'

# Tooltip

Displays a tooltip on mouse over.

## Default

<Playground>
  <Box display="flex" justifyContent="space-evenly">
    <Box borderRadius={20} border="2px solid #333" p={4} width="max-content">
      <Tooltip tooltip="tooltip">Tooltip around the text</Tooltip>
    </Box>
    <Tooltip tooltip="tooltip">
      <Box borderRadius={20} border="2px solid #333" p={4} width="max-content">
        Tooltip around the box
      </Box>
    </Tooltip>
  </Box>
</Playground>

## Place

Set place using `place` prop. 

<Playground>
  <Box display="flex" justifyContent="space-evenly">
    {['left', 'top', 'bottom', 'right'].map(place => 
      <Box key={place} p={2}>
        <Tooltip tooltip="tooltip" place={place}>{place}</Tooltip>
      </Box>
    )}
  </Box>
</Playground>

## Type

Set type using `type` prop. 

<Playground>
  <Box display="flex" justifyContent="space-between">
    {['dark', 'success', 'warning', 'error', 'info', 'light'].map(type => 
      <Box key={type} p={2}>
        <Tooltip tooltip="tooltip" type={type}>Type {type}</Tooltip>
      </Box>
    )}
  </Box>
</Playground>
